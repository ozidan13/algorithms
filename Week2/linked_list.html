class Node {
  constructor(value) {
    this.value = value;
    this.next = null;
  }
}

class LinkedList {
  constructor() {
    this.head = null;
  }
  
  // O(1) - Add to head
  prepend(value) {
    const node = new Node(value);
    node.next = this.head;
    this.head = node;
  }
  
  // O(n) - Add to tail
  append(value) {
    const node = new Node(value);
    if (!this.head) {
      this.head = node;
      return;
    }
    let curr = this.head;
    while (curr.next) curr = curr.next;
    curr.next = node;
  }
  
  // O(1) - Remove from head
  removeHead() {
    if (this.head) this.head = this.head.next;
  }
}
